                function collatzProof(n){
                    let results = [];
                    for(let i=1; i<=n; i++){
                        let r = collatzNum(i);
                        results.push(r);
                    }
                    return results;
                }

                function collatzNum(n){
                    let seq = [];
                    seq.push(n);
                    let f;
                    let r = n;
                    while(r != 1){
                      if(r%2 == 0){
                        f = (n) => n/2;
                      }
                      else{
                        f = (n) => 3*n + 1;
                      }
                      r = f(r);
                      seq.push(r);
                    }
                    return seq;
                  }
