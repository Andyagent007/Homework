  3. Use p5js to build something 3D that you haven’t done before. It could be a rotating donut, or a cat textured onto a cube … whatever creative idea you might have.

    answer

                        
                        function setup(){
                            createCanvas(windowWidth,windowHeight);
                            noStroke()
                            const flowerCount = 100;
                            for(let i=0; i < flowerCount; i++){ 
                                flower(randomColor(), { x: Math.random()*windowWidth, y : Math.random()*windowHeight });
                              }
                            }
                            
                            function flower(color, center){
                              const size = windowHeight/10
                              petals(7+Math.floor(Math.random()*5), size, color, center)
                              fill('yellow')
                              circle(center.x, center.y, size)
                            }
                      
                            function petals(n, size, color, center){
                              for(let i=0; i < n; i++){
                                let angle = (i/n)*Math.PI*2
                                let radius = size*0.8
                                let x = Math.cos(angle)*radius
                                let y = Math.sin(angle)*radius
                                fill(color)
                                circle(center.x + x, center.y + y, size)
                              }
                            }
                      
                            function randomColor(){
                              let r = random(255)
                              let g = random(255)
                              let b = random(255)
                              let a = random(200, 255)
                              return color(r, g, b, a)
                            }
                      
                            function coloredCircle(size, color){
                              c = {r: c.r, g: c.g - 1, b: c.b -2};
                              fill(c.r, c.g, c.b);
                              if(color){
                                fill(color);
                              }
                              circle(windowWidth/2, windowHeight/2, size);
                              if(c.r < 1 || c.g < 1 || c.b < 1){
                                c = {r: 225, g: 255, b: 255};
                              }
                            }
